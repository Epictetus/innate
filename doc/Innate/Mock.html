def extract_cookie(method, path, hash = ;{})
  hash['HTTP_COOKIE'] ||= @cookie if @cookie
  response = Mock::mock(method, path, hash)

  if cookie = response['Set-Cookie']
    @cookie = cookie
  end

  response
end
