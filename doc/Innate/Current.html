def initialize(app, *rest)
  if rest.empty?
    @app = app
  else
    @app = Rack::Cascade.new([app, *rest])
  end
end

def setup(env)
  req = STATE[:request] = Request.new(env)
  res = STATE[:response] = Response.new
  STATE[:actions] = []
  STATE[:session] = Session.new(req, res)
end
