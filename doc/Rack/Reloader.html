Rack::Reloader::Hooks

High performant source reloader

This class acts as Rack middleware.

It does not depend on Ramaze itself, but you might have to adjust the Reloader::Hooks module or include your own module to override the hooks. You also might have to set the Log constant.

What makes it especially suited for use in a production environment is that any file will only be checked once and there will only be made one system call stat(2).

Please note that this will not reload files in the background, it does so only when actively called In case of Ramaze it is performing a check/reload cycle at the start of every request, but also respects a cool down time, during which nothing will be done.

After every reload the OPTIONS hash will be checked for changed options and assigned to the instance, so you may change options during the lifetime of your application.

A number of hooks will be executed during the reload cycle, see Ramaze::ReloaderHooks for more information.

def call(env)
  options_reload

  if @cooldown and Time.now > @@last + @cooldown
    if @control
      instance_eval(&@control)
    elsif @thread
      Thread.exclusive{ cycle }
    else
      cycle
    end

    @@last = Time.now
  end

  @app.call(env)
end

def cycle
  before_cycle

  rotation do |file, stat|
    if mtime = stat.mtime
      if mtime > (@@mtimes[file] ||= mtime)
        safe_load(file, mtime)
      end
    else
      @@cache.delete(file)
    end
  end

  after_cycle
end

def figure_path(file, paths)
  if cached = @@cache[file]
    stat = ::File.stat(cached)
    return cached, stat if stat.file?
  elsif Pathname.new(file).absolute?
    stat = ::File.stat(file)
    return file, stat if stat.file? # do directories really end up in $" ?
  end

  paths.each do |possible_path|
    path = ::File.join(possible_path, file)

    begin
      stat = ::File.stat(path)
      return path, stat if stat.file?
    rescue Errno::ENOENT, Errno::ENOTDIR
    end
  end

  return nil
rescue Errno::ENOENT => ex
  @@cache.delete(file)
  retry
end

def rotation
  files = [$0, *$LOADED_FEATURES].uniq
  paths = ['./', *$LOAD_PATH].uniq

  files.each do |file|
    next if file =~ @ignore
    path, stat = figure_path(file, paths)

    if path and stat
      @@cache[file] = path
      yield(path, stat)
    else
      # Mostly harmless
    end
  end
end

def safe_load(file, mtime)
  before_safe_load(file)
  load(file)
  after_safe_load_succeed(file)
rescue Object => ex
  puts ex
  after_safe_load_failed(file, ex)
ensure
  @@mtimes[file] = mtime
end
